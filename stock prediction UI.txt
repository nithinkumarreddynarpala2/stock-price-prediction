        // --- HELPER WRAPPER FOR LUCIDE ---
        const Icon = ({ name, className = "w-5 h-5", ...props }) => {
            const ref = useRef(null);
            useEffect(() => {
                if (ref.current) {
                    ref.current.innerHTML = ''; 
                    const iconEl = document.createElement('i');
                    iconEl.setAttribute('data-lucide', name);
                    iconEl.className = className;
                    ref.current.appendChild(iconEl);
                    lucide.createIcons();
                }
            }, [name, className]);
            return <span ref={ref} {...props} className="inline-flex items-center justify-center"></span>;
        };

        const SimpleLineChart = ({ data, color, title, subtitle }) => {
            if (!data || data.length === 0) return null;
            const max = Math.max(...data);
            const min = Math.min(...data);
            const range = max - min || 1;
            const padding = range * 0.1; 
            const paddedMax = max + padding;
            const paddedMin = min - padding;
            const paddedRange = paddedMax - paddedMin;
            const points = data.map((val, i) => {
                const x = (i / (data.length - 1)) * 100;
                const y = 100 - ((val - paddedMin) / paddedRange) * 100;
                return `${x},${y}`;
            }).join(' ');

            return (
                <div className="bg-white dark:bg-slate-800 p-4 rounded-xl border border-slate-200 dark:border-slate-700 w-full flex flex-col h-64 shadow-lg relative overflow-hidden transition-all duration-300">
                    <div className="absolute inset-0 flex flex-col justify-between p-4 pointer-events-none opacity-20">

                    </div>
                    <div className="mb-4 z-10 relative">
                        <h3 className="text-slate-800 dark:text-slate-200 font-bold">{title}</h3>
                        {subtitle && <p className="text-slate-500 dark:text-slate-400 text-sm">{subtitle}</p>}
                    </div>
                    <div className="flex-grow relative w-full h-full z-10">
                        <svg viewBox="0 0 100 100" preserveAspectRatio="none" className="w-full h-full overflow-visible">
                            <defs>
                                <linearGradient id={`grad-${title.replace(/\s/g, '')}`} x1="0" x2="0" y1="0" y2="1">
                                    <stop offset="0%" stopColor={color} stopOpacity="0.3" />
                                    <stop offset="100%" stopColor={color} stopOpacity="0.0" />
                                </linearGradient>
                            </defs>
                            <polygon fill={`url(#grad-${title.replace(/\s/g, '')})`} points={`0,100 ${points} 100,100`} />
                            <polyline fill="none" stroke={color} strokeWidth="3" strokeLinecap="round" strokeLinejoin="round" points={points} className="drop-shadow-md" />
                        </svg>
                    </div>
                    <div className="absolute top-12 right-4 text-xs font-mono text-slate-500 pointer-events-none opacity-70">${max.toFixed(2)}</div>
                    <div className="absolute bottom-4 right-4 text-xs font-mono text-slate-500 pointer-events-none opacity-70">${min.toFixed(2)}</div>
                </div>
            );
        };
